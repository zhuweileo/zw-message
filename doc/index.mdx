# message

## Basic usage

<Playground>
    <div>
    {
        ['success', 'warn', 'info', 'error'].map(type => {
            return <button key={type} onClick={() => { message[type]('hahahahah') }}>{type}</button>
        })
    }
    </div>
</Playground>

## config

<Playground>
    {
        () => {
            message.config({
                maxCount: 5,
            })
            return <button onClick={() => message.info('maxCount 5')}>maxCount 5</button>
        }
    }
</Playground>


## duration

<Playground>
    {
        () => {
            return <div>
                <button onClick={() => message.info('duration 5s', 5)}>duration 5s</button>
                <button onClick={() => message.info('duration 1s', 1)}>duration 1s</button>
            </div> 
        }
    }
</Playground>

## removeCallback
<Playground>
    {
        () => {
            function removeCallback() {
                alert('removed')
            }
            return <div>
                <button onClick={() => message.info('alert callback', removeCallback)}>duration 5s</button>
            </div> 
        }
    }
</Playground>
